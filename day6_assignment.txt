create database dbday6work
use dbday6work

create table employee(empno int constraint pk_empno primary key,empname varchar(30),salary varchar(20),deptname varchar(30),bossno int)
create table departmnt(deptname varchar(30) constraint pk_deptname primary key,deptfloor int,deptphone int, mgrid int)
create table salees(salesno int constraint pk_salesno primary key,saleqty int, itemname varchar(40),
  deptname varchar(30))

create table employee(empno int constraint pk_empno primary key,empname varchar(30),salary varchar(20),deptname varchar(30) constraint fk_deptname 
  foreign key references departmnt(deptname) null,bossno int constraint fk_empno foreign key references employee(empno) null)

create table departmnt(deptname varchar(30) constraint pk_deptname primary key,deptfloor int,deptphone int, 
  mgrid int constraint fk_mgrid foreign key references employee(empno))

create table salees(salesno int constraint pk_salesno primary key,saleqty int, itemname varchar(40) 
  constraint fk_itemname foreign key references iteem(itemname),
  deptname varchar(30) constraint fk_deptname foreign key references departmnt(deptname))

create table iteem(itemname varchar(40) constraint pk_itemname primary key, itemtype char, itemcolor varchar(15))
sp_help employee
sp_help departmnt
sp_help salees
sp_help iteem
